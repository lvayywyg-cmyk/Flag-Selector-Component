3. 初始化组件

const flagDisplay = initFlagSelector({
    containerSelector: '.flag-placeholder',
    onSelect: function(selectedCountry) {
        console.log('选择的国家:', selectedCountry);
        // 处理国家选择逻辑
    }
});
